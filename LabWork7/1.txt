import os

# Запрос имени файла у пользователя
file_name = input("Введите имя файла: ")

if os.path.exists(file_name):
    while True:
        print("Выберите операцию:")
        print("1. Вывод содержимого файла")
        print("2. Удалить файл")
        print("3. Переименовать файл")

        choice = input("Введите номер операции: ")

        if choice == "1":
            with open(file_name, 'r') as file:
                print(file.read())
        elif choice == "2":
            os.remove(file_name)
            print(f"Файл {file_name} удален.")
            break  # После удаления файла, программа завершится
        elif choice == "3":
            new_name = input("Введите новое имя файла: ")
            os.rename(file_name, new_name)
            file_name = new_name  # Обновляем имя файла
            print(f"Файл переименован в {new_name}.")
        else:
            print("Неверный выбор.")
else:
    print("Файл не найден.")
