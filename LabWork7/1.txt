import os

file_name = input("Введите имя файла: ")

if os.path.exists(file_name):
    numbers_found = False
    with open(file_name, 'r') as file:
        for line in file:
            try:
                numbers = list(map(int, line.split()))
                if numbers:
                    numbers_found = True
                    total = 0  # Инициализация переменной для суммы
                    for num in numbers:
                        total += num  # Суммирование чисел вручную
                    print(f"Сумма чисел в строке с содержимым '{line.strip()}' : {total}")
            except ValueError:
                continue

    if not numbers_found:
        print("В файле нет чисел.")
else:
    print("Файл не найден.")
