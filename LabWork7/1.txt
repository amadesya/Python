import re

# Регулярное выражение для разделения на предложения
text = "Как дела? У меня всё хорошо... Надо будет встретиться! Пока."

# Разделяем по точке, восклицательному знаку, вопросительному знаку или многоточию
sentences = re.split(r'[.!?…]', text)

# Выводим каждое предложение
for sentence in sentences:
    sentence = sentence.strip()  # Убираем лишние пробелы
    if sentence:  # Игнорируем пустые строки
        print(sentence)
