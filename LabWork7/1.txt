import re

def replace_words(text):
    pattern = r'\b(редиск[аиуеойокамахами]{0,3}|нехорош[ийегоемуимем]{0,4} человек[ауомеиов]{0,4})\b'

    return re.sub(pattern, "*давайте жить дружно*", text, flags=re.IGNORECASE)

text = ("Редиска и нехороший человек в разных склонениях."
        "\nИменительный падеж: редиска, редиски, нехороший человек, нехорошие человеки"
        "\nРодительный падеж: редиски, редисок, нехорошего человека, нехороших человеков" 
        "\nВинительный падеж: редиска, редиски, нехорошего человека, нехороших человеков"
        "\nДательный падеж: редиске, редискам, нехорошему человеку, нехорошим человекам"
        "\nТворительный падеж редиской, редисками, нехорошим человеком, нехорошими человеками"
        "\nПредложный падеж: о редиске, о редисках, о нехорошем человеке, о нехороших человеках")

print(replace_words(text))


import re

def extract_dates(text):
    # Паттерн для поиска дат
    pattern = r'\b(\d{1,2})\.(\d{1,2})\.(\d{2,4})\b'
    
    # Поиск всех совпадений
    return re.findall(pattern, text)

# Пример использования
text = "Сегодня 13.03.2025, а вчера было 2.3.21. Также есть дата 05.12.2024."
print(extract_dates(text))


import re

def check_password():
    # Паттерн для проверки пароля
    pattern = r'^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)[A-Za-z\d]{6,}$'
    
    while True:
        password = input("Введите пароль: ")
        if re.search(pattern, password):
            print("Пароль принят.")
            break
        else:
            print("Пароль не соответствует требованиям. Попробуйте снова.")

# Пример использования
check_password()
