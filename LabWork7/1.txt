# Запрос имени файла у пользователя
file_name = input("Введите имя файла: ")

# Проверка существования файла
if os.path.exists(file_name):
    with open(file_name, 'r') as file:
        lines = file.readlines()
        
        # Напечатать первую строку
        if len(lines) > 0:
            print("Первая строка:", lines[0].strip())
        
        # Напечатать пятую строку
        if len(lines) > 4:
            print("Пятая строка:", lines[4].strip())
        
        # Напечатать первые 5 строк
        print("Первые 5 строк:")
        for line in lines[:5]:
            print(line.strip())
        
        # Запрос у пользователя s1 и s2 для вывода строк с s1-й по s2-ю
        s1 = int(input("Введите начальную строку (s1): "))
        s2 = int(input("Введите конечную строку (s2): "))
        print(f"Строки с {s1}-й по {s2}-ю:")
        for line in lines[s1-1:s2]:
            print(line.strip())
        
        # Напечатать весь файл
        print("Весь файл:")
        for line in lines:
            print(line.strip())
else:
    print("Файл не найден.")
