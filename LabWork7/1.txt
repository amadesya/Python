# Запрос имени файла у пользователя
file_name = input("Введите имя файла: ")

# Запрос у пользователя, перезаписать или дописать в файл
mode = input("Файл существует. Хотите перезаписать (w) или дописать (a)? ")

# Открытие файла в соответствующем режиме
with open(file_name, mode) as file:
    print("Введите строки. Введите 'end' для завершения.")
    
    # Цикл для ввода строк
    while True:
        line = input()
        if line == "end":
            break
        file.write(line + '\n')
