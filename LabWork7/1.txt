import os

file_name = input("Введите имя файла: ")

if os.path.exists(file_name):
    numbers_found = False
    with open(file_name, 'r') as file:
        if not numbers_found:
            print("В файле нет чисел.")
        else:
            for line in file:
                numbers = list(map(int, line.split(), errors='ignore'))
                if numbers:
                    numbers_found = True
                    print(f"Сумма чисел в строке: {sum(numbers)}")
else:
    print("Файл не найден.")
